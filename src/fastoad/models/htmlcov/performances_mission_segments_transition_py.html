28class DummyTransitionSegment(FlightSegment, mission_file_keyword="transition"): 

29 """ 

30 Computes a transient flight part in a very quick and dummy way. 

31 

32 :meth:`compute_from` will return only 2 or 3 flight points. 

33 

34 The second flight point is the end of transition and its mass is the start mass 

35 multiplied by :attr:`mass_ratio`. Other parameters are equal to those provided in 

36 :attr:`~fastoad.models.performances.mission.segments.base.FlightSegment.target`. 

37 

38 If :attr:`reserve_mass_ratio` is non-zero, a third flight point, with parameters equal 

39 to flight_point(2), except for mass where: 

40 mass(2) - reserve_mass_ratio * mass(3) = mass(3). 

41 In different words, mass(3) would be the Zero Fuel Weight (ZFW) and reserve can be 

42 expressed as a percentage of ZFW. 

43 """ 

44

61 def compute_from(self, start: FlightPoint) -> pd.DataFrame: 

62 

63 self.complete_flight_point(start) 

64 end = deepcopy(start) 

65 

66 end.mass = start.mass * self.mass_ratio 

67 end.altitude = self.target.altitude 

68 end.ground_distance = start.ground_distance + self.target.ground_distance 

69 end.mach = self.target.mach 

70 end.true_airspeed = self.target.true_airspeed 

71 end.equivalent_airspeed = self.target.equivalent_airspeed 

72 end.name = self.name 

73 self.complete_flight_point(end) 

74 

75 flight_points = [start, end] 

76 

77 if self.reserve_mass_ratio > 0.0: 

78 reserve = deepcopy(end) 

79 reserve.mass = end.mass / (1.0 + self.reserve_mass_ratio) 

80 flight_points.append(reserve) 

81 

82 return pd.DataFrame(flight_points)
