58class ValidityDomainChecker: 

59 """ 

60 Decorator class that checks variable values against limit bounds 

61 

62 This class aims at producing a status of out of limits variables at the end 

63 of an OpenMDAO computation. 

64 

65 The point is to allow to define limit bounds when defining an OpenMDAO 

66 system, but to make the check on the OpenMDAO problem after the run. 

67 

68 When defining an OpenMDAO system, use this class as Python decorator to 

69 define validity domains: 

70 

71 .. code-block:: 

72 

73 @ValidityDomainChecker 

74 class MyComponent(om.explicitComponent): 

75 ... 

76 

77 The above code will check values against lower and upper bounds that have 

78 been defined when adding OpenMDAO outputs. 

79 

80 Next code shows how to define lower and upper bounds, for inputs and/or 

81 outputs. 

82 

83 .. code-block:: 

84 

85 @ValidityDomainChecker( 

86 { 

87 "a:variable:with:two:bounds": (-10.0, 1.0), 

88 "a:variable:with:lower:bound:only": (0.0, None), 

89 "a:variable:with:upper:bound:only": (None, 4.2), 

90 }, 

91 ) 

92 class MyComponent(om.explicitComponent): 

93 ... 

94 

95 The defined domain limits supersedes lower and upper bounds from 

96 OpenMDAO output definitions, but only in the frame of ValidityDomainChecker. 

97 In any case, OpenMDAO process is not affected by usage of ValidityDomainChecker. 

98 

99 Validity status can be obtained through log messages from Python logging module 

100 after problem has been run with: 

101 

102 .. code-block:: 

103 

104 ... 

105 problem.run_model() 

106 ValidityDomainChecker.check_problem_variables(problem) 

107 

108 **Warnings**: 

109 - Units of limit values defined in ValidityDomainChecker are assumed to be the 

110 same as in add_input() and add_output() statements of decorated class 

111 - Validity check currently only applies to scalar values 

112 """ 

113 

114 _limit_definitions: Dict[UUID, "_LimitDefinitions"] = {} 

115 

116 def __init__(self, limits: Dict[str, tuple] = None, logger_name: str = None): 

117 """ 

118 :param limits: a dictionary where keys are variable names and values are two-values tuples 

119 that give lower and upper bound. One bound can be set to None. 

120 :param logger_name: The named of the logger that will be used. If not provided, name of 

121 current module (i.e. "__name__"") will be used. 

122 """ 

123 self._uuid = uuid.uuid4() 

124 

125 # use given logger name for now. If it is None, it will be replaced by 

126 # calling module name in __call__ (because I could not find a way to get 

127 # this module name here with inspect, though it is very easy when we 

128 # have the decorated class) 

129 limit_definitions = _LimitDefinitions(self._get_caller_filename(), logger_name) 

130 for var_name, bounds in limits.items(): 

131 lower = bounds[0] if bounds[0] is not None else -np.inf 

132 upper = bounds[1] if bounds[1] is not None else np.inf 

133 limit_definitions[var_name] = _LimitDefinition(lower, upper) 

134 

135 self._limit_definitions[self._uuid] = limit_definitions 

136 

137 def __call__(self, om_class: type): 

138 # Update logger name if needed: if it was not given, module name of 

139 # decorated class is used. 

140 if not self._limit_definitions[self._uuid].logger_name: 

141 self._limit_definitions[self._uuid].logger_name = om_class.__module__ 

142 

143 # We add to the OpenMDAO class a reference to self._limit_definitions[self._uuid] 

144 # to be able to update it with OpenMDAO declarations after problem setup. 

145 # See _update_problem_limit_definitions() 

146 om_class._fastoad_limit_definitions = self._limit_definitions[ 

147 self._uuid 

148 ] # pylint: disable=protected-access # We create it 

149 return om_class

151 @classmethod 

152 def check_problem_variables(cls, problem: om.Problem) -> List[CheckRecord]: 

153 """ 

154 Checks variable values in provided problem. 

155 

156 Logs warnings for each variable that is out of registered limits. 

157 

158 problem.setup() must have been run. 

159 

160 :param problem: 

161 :return: the list of checks 

162 """ 

163 

164 cls._update_problem_limit_definitions(problem) 

165 

166 variables = VariableList.from_problem(problem) 

167 records = cls.check_variables(variables) 

168 cls.log_records(records) 

169 return records

171 @classmethod 

172 def check_variables(cls, variables: VariableList) -> List[CheckRecord]: 

173 """ 

174 Check values of provided variables against registered limits. 

175 

176 :param variables: 

177 :return: the list of checks 

178 """ 

179 records: List[CheckRecord] = [] 

180 

181 for var in variables: 

182 for limit_definitions in cls._limit_definitions.values(): 

183 if var.name in limit_definitions: 

184 limit_def = limit_definitions[var.name] 

185 value = convert_units(var.value, var.units, limit_def.units) 

186 if value < limit_def.lower: 

187 status = ValidityStatus.TOO_LOW 

188 limit = limit_def.lower 

189 elif value > limit_def.upper: 

190 status = ValidityStatus.TOO_HIGH 

191 limit = limit_def.upper 

192 else: 

193 status = ValidityStatus.OK 

194 limit = None 

195 

196 records.append( 

197 CheckRecord( 

198 var.name, 

199 status, 

200 limit, 

201 limit_def.units, 

202 var.value, 

203 var.units, 

204 limit_definitions.source_file, 

205 limit_definitions.logger_name, 

206 ) 

207 ) 

208 return records

210 @staticmethod 

211 def log_records(records: List[CheckRecord]): 

212 """ 

213 Logs warnings through Python logging module for each CheckRecord in provided list if 

214 it is not OK. 

215 

216 :param records: 

217 :return: 

218 """ 

219 for record in records: 

220 if record.status != ValidityStatus.OK: 

221 logger = logging.getLogger(record.logger_name) 

222 limit_text = "under lower" if record.value < record.limit_value else "over upper" 

223 logger.warning( 

224 'Variable "%s" out of bound: value %s%s is %s limit ( %s%s ) in file %s', 

225 record.variable_name, 

226 record.value, 

227 " " + record.value_units if record.value_units else "", 

228 limit_text, 

229 record.limit_value, 

230 " " + record.limit_units if record.limit_units else "", 

231 record.source_file, 

232 )

234 @classmethod 

235 def _update_problem_limit_definitions(cls, problem: om.Problem): 

236 """ 

237 Updates limit definitions using variable declarations of provided OpenMDAO system. 

238 

239 problem.setup() must have been run. 

240 

241 :param problem: 

242 """ 

243 

244 variables = VariableList.from_problem( 

245 problem, get_promoted_names=False, promoted_only=False 

246 ) 

247 for var in variables: 

248 system_path = var.name.split(".") 

249 system = problem.model 

250 for system_name in system_path[:-1]: 

251 system = getattr(system, system_name) 

252 var_name = system_path[-1] 

253 

254 if hasattr(system, "_fastoad_limit_definitions"): 

255 

256 limit_definitions = system._fastoad_limit_definitions 

257 

258 if var_name in limit_definitions: 

259 # Get units for already defined limits 

260 limit_def = limit_definitions[var_name] 

261 if limit_def.units is None and var.units is not None: 

262 limit_def.units = var.units 

263 elif "lower" in var.metadata or "upper" in var.metadata: 

264 # Get bounds if defined in add_output. 

265 lower = var.metadata.get("lower") 

266 # lower can be None if it is not found OR if it defined and set to None 

267 if lower is None: 

268 lower = -np.inf 

269 upper = var.metadata.get("upper") 

270 # upper can be None if it is not found OR if it defined and set to None 

271 if upper is None: 

272 upper = np.inf 

273 units = var.metadata.get("units") 

274 if lower > -np.inf or upper < np.inf: 

275 limit_definitions[var_name] = _LimitDefinition(lower, upper, units)

277 @staticmethod 

278 def _get_caller_filename(): 

279 current_frame = inspect.currentframe() 

280 try: 

281 frame = inspect.getouterframes(current_frame)[2] 

282 filename = frame[1] 

283 finally: 

284 del current_frame 

285 del frame 

286 

287 return filename
