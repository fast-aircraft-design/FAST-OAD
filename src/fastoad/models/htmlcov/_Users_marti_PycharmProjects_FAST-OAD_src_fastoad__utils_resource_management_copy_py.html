27def copy_resource(package: Package, resource: str, target_path): 

28 """ 

29 Copies the indicated resource file to provided target path. 

30 

31 If target_path matches an already existing folder, resource file will be copied in this folder 

32 with same name. Otherwise, target_path will be the used name of copied resource file 

33 

34 :param package: package as in importlib.resources.read_binary() 

35 :param resource: resource as in importlib.resources.read_binary() 

36 :param target_path: file system path 

37 """ 

38 make_parent_dir(target_path) 

39 

40 with path(package, resource) as source_path: 

41 shutil.copy(source_path, target_path)

44def copy_resource_folder(package: Package, destination_path: str, exclude: List[str] = None): 

45 """ 

46 Copies the full content of provided package in destination folder. 

47 

48 Names of files that should not be copied have to be listed in `exclude`. 

49 

50 .. Warning :: 

51 

52 As the resources in the folder are discovered by browsing 

53 through the folder, they are not explicitly listed in the Python code. 

54 Therefore, to have the install process run smoothly, these resources need 

55 to be listed in the MANIFEST.in file. 

56 

57 :param package: name of resource package to copy 

58 :param destination_path: file system path of destination 

59 :param exclude: list of item names that should not be copied 

60 """ 

61 exclusion_list = ["__pycache__"] 

62 if exclude: 

63 exclusion_list += exclude 

64 

65 package_contents = PackageReader(package).contents 

66 if package_contents: 

67 for resource_name in package_contents: 

68 if resource_name in exclusion_list: 

69 continue 

70 if is_resource(package, resource_name): 

71 destination_file_path = pth.join(destination_path, resource_name) 

72 copy_resource(package, resource_name, destination_file_path) 

73 else: 

74 # In case of subfolders that are only declared in MANIFEST.in, 

75 # getattr(package, "resource_name") will fail (is there another way?). 

76 # So we fall back to using package name as as string. 

77 if isinstance(package, ModuleType): 

78 package_name = package.__name__ 

79 else: # str 

80 package_name = package 

81 new_package_name = ".".join([package_name, resource_name]) 

82 new_destination_path = pth.join(destination_path, resource_name) 

83 copy_resource_folder(new_package_name, new_destination_path, exclude=exclude)
