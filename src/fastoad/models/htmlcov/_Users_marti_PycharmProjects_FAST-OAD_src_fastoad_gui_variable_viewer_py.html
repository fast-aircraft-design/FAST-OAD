123 def load_variables(self, variables: VariableList, attribute_to_column: Dict[str, str] = None): 

124 """ 

125 Loads provided variable list and replace current data set. 

126 

127 :param variables: the variables to load 

128 :param attribute_to_column: dictionary keys tell what variable attributes are kept 

129 and the values tell what name will be displayed. If 

130 not provided, default translation will apply. 

131 """ 

132 

133 if not attribute_to_column: 

134 attribute_to_column = self._DEFAULT_COLUMN_RENAMING 

135 

136 self.dataframe = ( 

137 variables.to_dataframe() 

138 .rename(columns=attribute_to_column)[attribute_to_column.values()] 

139 .reset_index(drop=True) 

140 ) 

141 

142 self.dataframe["I/O"] = NA 

143 self.dataframe.loc[self.dataframe["is_input"] == 1, "I/O"] = INPUT 

144 self.dataframe.loc[self.dataframe["is_input"] == 0, "I/O"] = OUTPUT 

145 self.dataframe.drop(columns=["is_input"], inplace=True)

147 def get_variables(self, column_to_attribute: Dict[str, str] = None) -> VariableList: 

148 """ 

149 

150 :param column_to_attribute: dictionary keys tell what columns are kept and the values 

151 tell what variable attribute it corresponds to. If not 

152 provided, default translation will apply. 

153 :return: a variable list from current data set 

154 """ 

155 if not column_to_attribute: 

156 column_to_attribute = { 

157 value: key for key, value in self._DEFAULT_COLUMN_RENAMING.items() 

158 } 

159 

160 dataframe = self.dataframe.copy() 

161 

162 dataframe["is_input"] = None 

163 dataframe.loc[dataframe["I/O"] == INPUT, "is_input"] = True 

164 dataframe.loc[dataframe["I/O"] == OUTPUT, "is_input"] = False 

165 dataframe.drop(columns=["I/O"], inplace=True) 

166 variables = VariableList.from_dataframe( 

167 dataframe[column_to_attribute.keys()].rename(columns=column_to_attribute) 

168 ) 

169 return variables

258 else:

373 def _find_submodules(df: pd.DataFrame, modules: List[str] = None) -> Set[str]: 

374 """ 

375 Search for submodules at root or provided modules. 

376 

377 To find the submodules the method analyzes the name of the variables. 

378 If the kwarg 'modules' is not None, the submodules search will be applied to 

379 the variables that are part of these modules. 

380 

381 :param df: the pandas dataframe containing the variables 

382 :param modules: the list of modules to which the variables belong 

383 :return the submodules list 

384 """ 

385 var_names = df.filter(items=["Name"]) 

386 

387 if not modules: 

388 modules = [] 

389 

390 def get_next_module(path): 

391 submodules = path.split(":") 

392 if len(modules) >= len(submodules) or submodules[: len(modules)] != modules: 

393 return "" 

394 

395 return submodules[len(modules)] 

396 

397 submodules = var_names.applymap(get_next_module) 

398 submodules = submodules[submodules.Name != ""] 

399 

400 return set(submodules["Name"].tolist())

402 @staticmethod 

403 def _filter_variables( 

404 df: pd.DataFrame, modules: List[str], var_io_type: str = None 

405 ) -> pd.DataFrame: 

406 """ 

407 Returns a filtered dataframe with respect to a set of modules and variable type. 

408 

409 The variables kept must be part of the modules list provided and the variable type 

410 'IN' or 'OUT'(if provided). 

411 

412 :param df: the pandas dataframe containing the variables 

413 :param modules: the list of modules to which the variables belong 

414 :param var_io_type: the type of variables to keep 

415 :return the filtered dataframe 

416 """ 

417 if var_io_type is None: 

418 var_io_type = TAG_ALL 

419 path = "" 

420 for _ in modules: 

421 if modules[-1] == TAG_ALL: 

422 path = ":".join(modules[:-1]) 

423 else: 

424 path = ":".join(modules) 

425 

426 path_filter = [True] * len(df) if path == "" else df.Name.str.startswith(path) 

427 io_filter = [True] * len(df) if var_io_type == TAG_ALL else df["I/O"] == var_io_type 

428 

429 filtered_df = df[path_filter & io_filter] 

430 

431 return filtered_df
