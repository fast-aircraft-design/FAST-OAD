146 def sellar_setup(sellar_instance: Sellar): 

147 """ 

148 Sets up the Sellar problem, given a Sellar Group() instance. 

149 

150 Pure OpenMDAO scripting. 

151 """ 

152 problem = Problem() 

153 problem.model = sellar_instance 

154 

155 problem.driver = ScipyOptimizeDriver() 

156 

157 problem.driver.options["optimizer"] = "SLSQP" 

158 problem.driver.options["tol"] = 1.0e-08 

159 # pb.driver.options['maxiter'] = 100 

160 problem.driver.options["disp"] = True 

161 

162 problem.model.add_design_var("x", lower=0, upper=10) 

163 problem.model.add_design_var("z", lower=0, upper=10) 

164 

165 problem.model.add_objective("f") 

166 

167 problem.model.add_constraint("g1", upper=0.0) 

168 problem.model.add_constraint("g2", upper=0.0) 

169 

170 problem.setup() 

171 

172 return problem
