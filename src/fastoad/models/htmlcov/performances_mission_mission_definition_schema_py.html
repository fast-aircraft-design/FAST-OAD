46class MissionDefinition(dict): 

47 def __init__(self, file_path: Union[str, PathLike] = None): 

48 """ 

49 Class for reading a mission definition from a YAML file. 

50 

51 Path of YAML file should be provided at instantiation, or in 

52 :meth:`load`. 

53 

54 :param file_path: path of YAML file to read. 

55 """ 

56 super().__init__() 

57 if file_path: 

58 self.load(file_path) 

59 

60 def load(self, file_path: Union[str, PathLike]): 

61 """ 

62 Loads a mission definition from provided file path. 

63 

64 Any existing definition will be overwritten. 

65 

66 :param file_path: path of YAML file to read. 

67 """ 

68 self.clear() 

69 yaml = YAML() 

70 

71 with open(file_path) as yaml_file: 

72 data = yaml.load(yaml_file) 

73 

74 with open_text(resources, JSON_SCHEMA_NAME) as json_file: 

75 json_schema = json.loads(json_file.read()) 

76 validate(data, json_schema) 

77 

78 self._validate(data) 

79 self.update(data)

81 @classmethod 

82 def _validate(cls, content: dict): 

83 """ 

84 Does a second pass validation of file content. 

85 

86 Also applies this feature: 

87 - polar: foo:bar 

88 is translated to: 

89 - polar: 

90 CL: foo:bar:CL 

91 CD: foo:bar:CD 

92 

93 Errors are raised if file content is incorrect. 

94 

95 :param content: 

96 """ 

97 phase_names = set(content[PHASE_DEFINITIONS_TAG].keys()) 

98 

99 for phase_definition in content[PHASE_DEFINITIONS_TAG].values(): 

100 cls._process_polar_definition(phase_definition) 

101 for segment_definition in phase_definition[PARTS_TAG]: 

102 cls._process_polar_definition(segment_definition) 

103 

104 for route_definition in content[ROUTE_DEFINITIONS_TAG].values(): 

105 cls._process_polar_definition(route_definition[CRUISE_PART_TAG]) 

106 for part in list(route_definition[CLIMB_PARTS_TAG]) + list( 

107 route_definition[DESCENT_PARTS_TAG] 

108 ): 

109 cls._process_polar_definition(part) 

110 Ensure(part[PHASE_TAG]).is_in(phase_names) 

111 

112 for mission_definition in content[MISSION_DEFINITION_TAG].values(): 

113 reserve_count = 0 

114 for part in mission_definition[PARTS_TAG]: 

115 part_type, value = tuple(*part.items()) 

116 if part_type == PHASE_TAG: 

117 Ensure(value).is_in(content[PHASE_DEFINITIONS_TAG].keys()) 

118 elif part_type == ROUTE_TAG: 

119 Ensure(value).is_in(content[ROUTE_DEFINITIONS_TAG].keys()) 

120 elif part_type == RESERVE_TAG: 

121 reserve_count += 1 

122 Ensure(value["ref"]).is_in(content[ROUTE_DEFINITIONS_TAG].keys()) 

123 Ensure(reserve_count).is_less_than_or_equal_to(1) 

124 if reserve_count == 1: 

125 # reserve definition should be the last part 

126 Ensure(part_type).equals(RESERVE_TAG) 

127 

128 @staticmethod 

129 def _process_polar_definition(struct: dict): 

130 """ 

131 If "foo:bar:baz" is provided as value for the "polar" key in provided dictionary, it is 

132 replaced by the dict {"CL":"foo:bar:baz:CL", "CD":"foo:bar:baz:CD"} 

133 """ 

134 if POLAR_TAG in struct: 

135 polar_def = struct[POLAR_TAG] 

136 if isinstance(polar_def, str) and ":" in polar_def: 

137 struct[POLAR_TAG] = OrderedDict({"CL": polar_def + ":CL", "CD": polar_def + ":CD"})
