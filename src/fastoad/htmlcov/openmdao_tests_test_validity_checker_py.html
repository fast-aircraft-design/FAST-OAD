37def set_logger_file(log_file_path): 

38 logger = logging.getLogger("main") 

39 for hdlr in logger.handlers: 

40 logger.removeHandler(hdlr) 

41 

42 formatter = logging.Formatter("%(name)s - %(message)s") 

43 handler = FileHandler(log_file_path) 

44 handler.setFormatter(formatter) 

45 logger.addHandler(handler)
