26def get_float_list_from_string(text: str): 

27 """ 

28 Parses the provided string and returns a list of floats if possible. 

29 

30 If provided text is not numeric, None is returned. 

31 

32 New line characters are simply ignored. 

33 

34 As an example, following patterns will result as list [1., 2., 3.] 

35 

36 .. code-block:: 

37 

38 '[ 1, 2., 3]' 

39 ' 1, 2., 3' 

40 ' 1 2 3' 

41 ' 1 2 3 dummy 4' 

42 """ 

43 

44 text_value = text.strip() 

45 if not text_value: 

46 return None 

47 

48 # If it begins by '[', an array is expected, potentially multidimensional 

49 if text_value.startswith("["): 

50 # The string is first transformed in a way that can be parsed by genfromtxt 

51 text_value = re.sub(r"\r?\n|\r", "", text_value) # first remove all new lines 

52 text_value = re.sub(r"\]\s*,\s*\[", "\n", text_value) 

53 text_value = text_value.strip("[]") 

54 text_io = io.StringIO(text_value) 

55 try: 

56 return np.genfromtxt(text_io, delimiter=",").tolist() 

57 except ValueError as exc: 

58 raise FastCouldNotParseStringToArrayError(text.strip(), exc) 

59 

60 # Deals with multiple values in same element. numpy.fromstring can parse a string, 

61 # but we have to test with either ' ' or ',' as separator. The longest result should be 

62 # the good one. 

63 try: 

64 value1 = np.fromstring(text_value, dtype=float, sep=" ").tolist() 

65 except ValueError: 

66 pass 

67 

68 try: 

69 value2 = np.fromstring(text_value, dtype=float, sep=",").tolist() 

70 except ValueError: 

71 pass 

72 

73 if not value1 and not value2: 

74 return None 

75 

76 return value1 if len(value1) > len(value2) else value2

79class FastCouldNotParseStringToArrayError(FastError): 

80 """Raised when a conversion from string to array failed."""

82 def __init__(self, parsed_text, original_exception): 

83 super().__init__() 

84 self.text = parsed_text

87 def __str__(self): 

88 msg = 'Could not parse "%s"' % self.text
